def _get_class(property_name: str) -> Property:

    ResourceLimit = iocage.Config.Jail.Properties.ResourceLimit
    DefaultrouterModule = iocage.Config.Jail.Properties.Defaultrouter

    if (property_name == "ip4_addr"):
        return iocage.Config.Jail.Properties.Addresses.IPv4AddressesProp
    elif (property_name == "ip6_addr"):
        return iocage.Config.Jail.Properties.Addresses.IPv6AddressesProp
    elif property_name == "interfaces":
        return iocage.Config.Jail.Properties.Interfaces.InterfaceProp
    elif property_name == "resolver":
        return iocage.Config.Jail.Properties.Resolver.ResolverProp
    elif property_name == "defaultrouter":
        return DefaultrouterModule.DefaultrouterProp
    elif property_name == "defaultrouter6":
        return DefaultrouterModule.Defaultrouter6Prop
    elif property_name == "depends":
        return iocage.Config.Jail.Properties.Depends.DependsProp
    elif property_name in ResourceLimit.properties:
        return ResourceLimit.ResourceLimitProp

    raise ValueError("A special property class with this name was not found")



[docs]def init_property(
    property_name: str,
    config: typing.Optional[
        'iocage.Config.Jail.JailConfig.JailConfig'
    ]=None,
    logger: typing.Optional['iocage.Logger.Logger']=None
) -> Property:  # noqa: T484
    """
    Instantiate a special jail config property.

    Args:

        property_name:
            The name of the special property that should be instantiated:
                - ip4_addr
                - ip6_addr
                - interfaces
                - resolver
                - defaultrouter
                - defaultrouter6
                - depends

        config (iocage.Config.Jail.JailConfig.JailConfig):
            The special property keeps reference to this JailConfig instance

        logger (iocage.Logger.Logger): (optional):
            Logger instance that the class is initialized with
    """
    target_class = _get_class(property_name)
    out = target_class(
        property_name=property_name,
        config=config,
        logger=logger
    )
    return out
